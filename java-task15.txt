// WAP to perform Addition, Subtraction, Multiplication, Division Operation using Method and ParameterizedÂ Constructor

package myData;

class Arithmetric {
	int num1,num2;
	
	Arithmetric(int n1,int n2)
	{
		num1 = n1;
		num2 = n2;
	}
	
	public void add()
	{
		int sum = num1 + num2;
		System.out.println(num1 + " + " + num2 + " = " + sum);
	}
	
	public void subtract()
	{
		int diff = num1 - num2;
		System.out.println(num1 + " - " + num2 + " = " + diff);
	}
	
	public void multiply()
	{
		int multi = num1 * num2;
		System.out.println(num1 + " X " + num2 + " = " + multi);
	}
	
	public void divide()
	{
		double div = (double)num1/num2;
		System.out.println(num1+ "/" + num2 + " = " + div);
	}
	
	
}

public class ArithmeticParaConstructor {

	public static void main(String[] args) {
		Arithmetric obj = new Arithmetric(9,3);
		obj.add();
		obj.subtract();
		obj.multiply();
		obj.divide();
}
}
